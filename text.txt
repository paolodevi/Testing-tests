scr = get(0, 'ScreenSize');
figWidth  = (2/3) * scr(3);
figHeight = (2/3) * scr(4);

% Center the figure on screen
figLeft = (scr(3) - figWidth) / 2;
figBottom = (scr(4) - figHeight) / 2;

% Create the figure
figure('Position', [figLeft, figBottom, figWidth, figHeight], ...
       'Color', 'w', ...
       'Name', 'Two-thirds Screen Figure', ...
       'NumberTitle', 'off');
