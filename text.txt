function S = importTextData(filepath, headerLine)
% IMPORTTEXTDATA Import column data from a text file with a header
%
% S = importTextData(filepath, headerLine)
%
% Inputs
% filepath   : path to the text file
% headerLine : line number where the header (#...) is located
%
% Output
% S : struct with fields named after variables in the header

%% Read file lines
fid = fopen(filepath,'r');
assert(fid ~= -1, 'Cannot open file');

lines = {};
while ~feof(fid)
    lines{end+1} = fgetl(fid);
end
fclose(fid);

%% Extract header
header = strtrim(lines{headerLine});
header = regexprep(header,'^#',''); % remove #

% Split variables
vars = strsplit(header,';');

nVar = numel(vars);
varNames = cell(1,nVar);

for i = 1:nVar
    
    % Remove units in []
    name = regexprep(vars{i},'\[.*?\]','');
    
    % Clean spaces
    name = strtrim(name);
    
    % Convert to valid MATLAB name
    name = matlab.lang.makeValidName(name);
    
    varNames{i} = name;
    
end

%% Detect delimiter from first data line
dataStart = headerLine + 1;

while dataStart <= numel(lines) && isempty(strtrim(lines{dataStart}))
    dataStart = dataStart + 1;
end

sampleLine = lines{dataStart};

if contains(sampleLine,';')
    delimiter = ';';
else
    delimiter = 'whitespace';
end

%% Read numeric data
if strcmp(delimiter,';')
    data = readmatrix(filepath,'Delimiter',';','NumHeaderLines',headerLine);
else
    data = readmatrix(filepath,'NumHeaderLines',headerLine);
end

%% Store in struct
S = struct;

for i = 1:nVar
    S.(varNames{i}) = data(:,i);
end

end
