nFrames = min(length(frames1), length(frames2));

figure;
for k = 1:nFrames
    % Extract frame images
    img1 = frames1(k).cdata;
    img2 = frames2(k).cdata;

    % If needed, resize to same height
    if size(img1,1) ~= size(img2,1)
        targetH = min(size(img1,1), size(img2,1));
        img1 = imresize(img1, [targetH NaN]);
        img2 = imresize(img2, [targetH NaN]);
    end

    % Concatenate horizontally
    combined = [img1, img2];

    % Display efficiently
    imshow(combined);
    title(sprintf('Frame %d / %d', k, nFrames));
    drawnow;
end
