% Find all open figures
figs = findall(groot, 'Type', 'figure');

for k = 1:numel(figs)

    fig = figs(k);

    % Get current axes in the figure
    ax = get(fig, 'CurrentAxes');

    % Default filename if no title exists
    filename = sprintf('Figure_%d', fig.Number);

    if ~isempty(ax)
        t = get(get(ax, 'Title'), 'String');

        % Handle title being a cell array
        if iscell(t)
            t = strjoin(t, '_');
        end

        if ~isempty(t)
            filename = t;
        end
    end

    % Replace spaces with underscores
    filename = strrep(filename, ' ', '_');

    % Remove characters invalid for filenames
    filename = regexprep(filename, '[<>:"/\\|?*]', '');

    % Save figure
    savefig(fig, [filename '.fig']);

    fprintf('Saved: %s.fig\n', filename);

end
