filename = 'yourfile.txt';

% Open file for reading
fid = fopen(filename, 'r');
if fid == -1
    error('Cannot open file.');
end

lines = {};
while ~feof(fid)
    line = fgets(fid);   % keeps newline characters
    if length(line) > 3
        line = line(4:end); % remove first three characters
    else
        line = ''; % line shorter than 3 characters
    end
    lines{end+1} = line; %#ok<SAGROW>
end
fclose(fid);

% Open file for writing (overwrite)
fid = fopen(filename, 'w');
if fid == -1
    error('Cannot write to file.');
end

for i = 1:numel(lines)
    fprintf(fid, '%s', lines{i});
end
fclose(fid);
