function printTree(tree)

    fprintf('%s\n', tree.name);

    for k = 1:numel(tree.children)
        isLast = (k == numel(tree.children));
        printNode(tree.children{k}, '', isLast);
    end

end


function printNode(node, prefix, isLast)

    if isLast
        connector = '└── ';
        nextPrefix = [prefix '    '];
    else
        connector = '├── ';
        nextPrefix = [prefix '│   '];
    end

    fprintf('%s%s%s\n', prefix, connector, node.name);

    for k = 1:numel(node.children)
        childIsLast = (k == numel(node.children));
        printNode(node.children{k}, nextPrefix, childIsLast);
    end

end
