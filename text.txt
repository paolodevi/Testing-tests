function Faces = build_faces_from_edges(Edges)
    % Very simple loop closure (works for clean loops only)
    used = false(size(Edges,1),1);
    Faces = {};
    for i = 1:size(Edges,1)
        if used(i), continue; end
        loop = Edges(i,:);
        used(i) = true;
        current = loop(end);
        % Follow connected edges to close loop
        while true
            [nextEdge, idx] = find(Edges == current);
            nextEdge = nextEdge(~used(nextEdge));
            if isempty(nextEdge)
                break;
            end
            used(nextEdge(1)) = true;
            e = Edges(nextEdge(1),:);
            current = e(e~=current);
            loop(end+1) = current;
            if loop(1) == loop(end)
                break;
            end
        end
        if loop(1) == loop(end)
            Faces{end+1} = loop(1:end-1);
        end
    end
    Faces = Faces'; % cell array of faces
end
